// Reverse the digits
int reverseDigits(int num) {
    int r,rev = 0;
    while (num > 0) {
    	r=num%10;
        rev = rev * 10+r;
        num = num /10;
    }
    return rev;
}


// Adams number
int isAdams(int num) {
    int sq,rev,r_sq,r_sq2;
	sq=num*num;
	rev=reverseDigits(num);
	r_sq=rev*rev;
	r_sq2=reverseDigits(r_sq);
	return(sq==r_sq2);
}


// Prime number
int isPrime(int num){
	if (num <= 1){
		return 0;
    }
	for (int i = 2; i <= num / 2; i++) {
		if (num% i == 0){		
			return 0;
		}
    }  
	return 1;
}


// Prime palindrome
int isPrimePalindrome(int num) {
